services:
  # ===========================
  # ИНФРАСТРУКТУРА
  # ===========================

  postgres:
    image: postgres:15-alpine
    container_name: postgres
    environment:
      POSTGRES_DB: ${POSTGRES_DB:-shop}
      POSTGRES_USER: ${POSTGRES_USER:-shop}
      POSTGRES_PASSWORD: ${POSTGRES_PASSWORD:-shop_password}
    volumes:
      - postgres_data:/var/lib/postgresql/data
      - ./infra/postgres-init:/docker-entrypoint-initdb.d:ro
    ports:
      - "5432:5432"
    networks:
      - microshop-net
    healthcheck:
      test: ["CMD-SHELL", "pg_isready -U ${POSTGRES_USER:-shop} -d ${POSTGRES_DB:-shop}"]
      interval: 10s
      timeout: 5s
      retries: 5

  redis:
    image: redis:7-alpine
    container_name: redis
    command: redis-server --appendonly yes
    volumes:
      - redis_data:/data
    ports:
      - "6379:6379"
    networks:
      - microshop-net
    healthcheck:
      test: ["CMD", "redis-cli", "ping"]
      interval: 10s
      timeout: 3s
      retries: 3

  rabbitmq:
    image: rabbitmq:3.12-management-alpine
    container_name: rabbitmq
    environment:
      RABBITMQ_DEFAULT_USER: ${RABBITMQ_USER:-guest}
      RABBITMQ_DEFAULT_PASS: ${RABBITMQ_PASSWORD:-guest}
    volumes:
      - rabbitmq_data:/var/lib/rabbitmq
    ports:
      - "5672:5672"
      - "15672:15672"
    networks:
      - microshop-net
    healthcheck:
      test: ["CMD", "rabbitmq-diagnostics", "check_running"]
      interval: 30s
      timeout: 10s
      retries: 3

  etcd:
    image: gcr.io/etcd-development/etcd:v3.5.10
    container_name: etcd
    command: >
      /usr/local/bin/etcd
      --name etcd0
      --listen-client-urls http://0.0.0.0:2379
      --advertise-client-urls http://etcd:2379
    ports:
      - "2379:2379"
    networks:
      - microshop-net
    healthcheck:
      test: ["CMD", "etcdctl", "endpoint", "health"]
      interval: 10s
      timeout: 5s
      retries: 3

  minio:
    image: minio/minio:latest
    container_name: minio-store
    environment:
      MINIO_ROOT_USER: ${MINIO_ROOT_USER:-minio_admin}
      MINIO_ROOT_PASSWORD: ${MINIO_ROOT_PASSWORD:-minio_password}
    command: server /data --console-address ":9001"
    volumes:
      - minio_data:/data
      - ./infra/minio-init:/seed:ro
    ports:
      - "9000:9000"
      - "9001:9001"
    networks:
      - microshop-net
    healthcheck:
      test: ["CMD", "mc", "ready", "local"]
      interval: 10s
      timeout: 5s
      retries: 3

  # ===========================
  # ПРИКЛАДНЫЕ СЕРВИСЫ
  # ===========================

  auth-service:
    build:
      context: ./auth-service-go
      dockerfile: Dockerfile
    container_name: auth-service
    environment:
      AUTH_PORT: ${AUTH_PORT:-8081}
      ETCD_ENDPOINTS: ${ETCD_ENDPOINTS:-http://etcd:2379}
      REDIS_ADDR: ${REDIS_HOST:-redis}:${REDIS_PORT:-6379}
      SESSION_TTL_SECONDS: ${AUTH_SESSION_TTL_SECONDS:-86400}
    ports:
      - "8081:8081"
    networks:
      - microshop-net
    depends_on:
      etcd:
        condition: service_healthy
      redis:
        condition: service_healthy
    restart: unless-stopped

  backend-service:
    build:
      context: ./backend-java
      dockerfile: Dockerfile
    container_name: backend-service
    environment:
      SPRING_DATASOURCE_URL: jdbc:postgresql://${POSTGRES_HOST:-postgres}:${POSTGRES_PORT:-5432}/${POSTGRES_DB:-shop}
      SPRING_DATASOURCE_USERNAME: ${POSTGRES_USER:-shop}
      SPRING_DATASOURCE_PASSWORD: ${POSTGRES_PASSWORD:-shop_password}
      SPRING_REDIS_HOST: ${REDIS_HOST:-redis}
      SPRING_REDIS_PORT: ${REDIS_PORT:-6379}
      SPRING_RABBITMQ_HOST: ${RABBITMQ_HOST:-rabbitmq}
      SPRING_RABBITMQ_PORT: ${RABBITMQ_PORT:-5672}
      SPRING_RABBITMQ_USERNAME: ${RABBITMQ_USER:-guest}
      SPRING_RABBITMQ_PASSWORD: ${RABBITMQ_PASSWORD:-guest}
      MINIO_ENDPOINT: ${MINIO_ENDPOINT:-http://minio:9000}
      MINIO_ACCESS_KEY: ${MINIO_ROOT_USER:-minio_admin}
      MINIO_SECRET_KEY: ${MINIO_ROOT_PASSWORD:-minio_password}
    ports:
      - "8080:8080"
    networks:
      - microshop-net
    depends_on:
      postgres:
        condition: service_healthy
      redis:
        condition: service_healthy
      rabbitmq:
        condition: service_healthy
      minio:
        condition: service_healthy
    restart: unless-stopped

  order-worker:
    build:
      context: ./order-worker-python
      dockerfile: Dockerfile
    container_name: order-worker
    environment:
      RABBITMQ_HOST: ${RABBITMQ_HOST:-rabbitmq}
      RABBITMQ_PORT: ${RABBITMQ_PORT:-5672}
      RABBITMQ_USER: ${RABBITMQ_USER:-guest}
      RABBITMQ_PASSWORD: ${RABBITMQ_PASSWORD:-guest}
      RABBITMQ_QUEUE: ${RABBITMQ_QUEUE:-orders.queue}
      POSTGRES_HOST: ${POSTGRES_HOST:-postgres}
      POSTGRES_PORT: ${POSTGRES_PORT:-5432}
      POSTGRES_DB: ${POSTGRES_DB:-shop}
      POSTGRES_USER: ${POSTGRES_USER:-shop}
      POSTGRES_PASSWORD: ${POSTGRES_PASSWORD:-shop_password}
    networks:
      - microshop-net
    depends_on:
      rabbitmq:
        condition: service_healthy
      postgres:
        condition: service_healthy
    restart: unless-stopped

  frontend:
    build:
      context: ./frontend-react
      dockerfile: Dockerfile
    container_name: frontend
    environment:
      VITE_API_BASE: ${VITE_API_BASE:-/api}
    networks:
      - microshop-net
    depends_on:
      - backend-service
      - auth-service
    restart: unless-stopped

  nginx-gateway:
    image: nginx:alpine
    container_name: nginx-gateway
    ports:
      - "${GATEWAY_PORT:-3000}:80"
    volumes:
      - ./nginx-gateway/nginx.conf:/etc/nginx/conf.d/default.conf:ro
    networks:
      - microshop-net
    depends_on:
      - frontend
      - backend-service
      - auth-service
      - minio
    restart: unless-stopped

networks:
  microshop-net:
    driver: bridge

volumes:
  postgres_data:
  redis_data:
  rabbitmq_data:
  minio_data:
